python 获取当前文件夹下的所有的文件名

发表于2019-04-29|更新于2024-09-30|python
|阅读量:
import os


def file_name(file_dir):
    for root, dirs, files in os.walk(file_dir):
        print(root)  # 当前目录路径
        print(dirs)  # 当前路径下所有子目录
        print(files)  # 当前路径下所有非目录子文件
